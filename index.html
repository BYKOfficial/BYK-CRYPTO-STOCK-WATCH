let allCoins = []; let watchlist = JSON.parse(localStorage.getItem("watchlist")) || []; // LOAD DATA async function loadCrypto(){ try{ const res = await fetch("https://api.coingecko.com/api/v3/coins/markets?vs_currency=usd"); const data = await res.json(); allCoins = data; showCoins(data.slice(0,20)); showStats(data); showNextBTC(data); showWatchlist(); detectHotCoins(data); document.getElementById("status").innerText = "✅ Auto Updated"; }catch{ document.getElementById("status").innerText = "❌ Reloading..."; setTimeout(loadCrypto,3000); } } // SHOW COINS function showCoins(coins){ let html=""; coins.forEach(c=>{ let hot = c.price_change_percentage_24h > 5 ? "🔥 HOT" : ""; html += `

${c.name} ${hot}

$${c.current_price}

`; }); document.getElementById("crypto").innerHTML = html; } // HOT COINS ALERT function detectHotCoins(data){ data.forEach(c=>{ if(c.price_change_percentage_24h > 10){ console.log("HOT COIN:", c.name); } }); } // AUTO SEARCH function filterCoins(){ const val = document.getElementById("search").value.toLowerCase(); const filtered = allCoins.filter(c=>c.name.toLowerCase().includes(val)); showCoins(filtered.slice(0,20)); } // WATCHLIST function addToWatch(name){ if(!watchlist.includes(name)){ watchlist.push(name); localStorage.setItem("watchlist", JSON.stringify(watchlist)); showWatchlist(); } } function showWatchlist(){ document.getElementById("watchlist").innerHTML = watchlist.map(w=>`

⭐ ${w}

`).join(""); } // STATS function showStats(data){ const gainers = [...data].sort((a,b)=>b.price_change_percentage_24h - a.price_change_percentage_24h).slice(0,5); const losers = [...data].sort((a,b)=>a.price_change_percentage_24h - b.price_change_percentage_24h).slice(0,5); document.getElementById("gainers").innerHTML = gainers.map(c=>`

🚀 ${c.name} (${c.price_change_percentage_24h.toFixed(2)}%)

`).join(""); document.getElementById("losers").innerHTML = losers.map(c=>`

📉 ${c.name} (${c.price_change_percentage_24h.toFixed(2)}%)

`).join(""); } // NEXT BTC function showNextBTC(data){ const coins = data.filter(c=>c.market_cap < 5000000000 && c.price_change_percentage_24h > 5).slice(0,5); document.getElementById("nextbtc").innerHTML = coins.map(c=>`

🔥 ${c.name}

`).join(""); } // AUTO THEME function autoTheme(){ const hour = new Date().getHours(); document.body.style.background = (hour > 6 && hour < 18) ? "white" : "#0d1117"; } // AUTO ALERT function notify(msg){ if(Notification.permission === "granted"){ new Notification(msg); } } if(Notification.permission !== "granted"){ Notification.requestPermission(); } // AUTO RUN setInterval(loadCrypto,15000); setInterval(autoTheme,60000); loadCrypto(); autoTheme();
